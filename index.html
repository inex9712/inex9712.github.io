AI Smart File Organizer

Python Gemini API pathlib shutil

프로젝트 개요

파일명과 확장자만으로 Gemini 2.5 Flash 모델이 카테고리를 추론하고, 생성 연도/카테고리 2뎁스 구조로 디렉토리를 자동 생성해 파일을 분류하는 Python 자동화 툴입니다.

핵심 기능

  • 🤖 AI 메타데이터 추론 — Gemini API로 파일 카테고리 자동 판별
  • 📁 동적 디렉토리 생성 — 연도 / 카테고리 2뎁스 자동 구조화
  • 🌐 다국어 파일명 지원 — UTF-8 강제 적용으로 한/일 파일명 처리
  • 🛡️ 자가 참조 버그 방지 — 실행 스크립트 자신을 이동 대상에서 제외

트러블슈팅

Python 3.14 환경에서 다국어 파일명 처리 시 ASCII Codec 에러 발생 → sys.stdout.reconfigure + 터미널 -X utf8 플래그 조합으로 딥 레벨 해결.

소스 코드

import os, sys
os.environ["PYTHONUTF8"] = "1"
if sys.stdout.encoding.lower() != 'utf-8':
    sys.stdout.reconfigure(encoding='utf-8')

import shutil, datetime
from pathlib import Path
from google import genai

client = genai.Client(api_key="YOUR_API_KEY")
BASE_DIR = Path(__file__).parent
TARGET_FOLDER = BASE_DIR / "test_folder"

def get_category_from_ai(file_name, extension):
    if not extension:
        return "기타_확장자없음"
    prompt = f"파일명: {file_name}\n확장자: {extension}\n폴더 이름 1개만 출력"
    try:
        response = client.models.generate_content(
            model='gemini-2.5-flash', contents=prompt)
        return response.text.strip().replace("/","_")
    except:
        return "기타_분류실패"

def organize_and_move_files():
    for item in TARGET_FOLDER.iterdir():
        if not item.is_file(): continue
        if item.name == Path(__file__).name: continue  # 자기 자신 제외
        year = datetime.datetime.fromtimestamp(
            item.stat().st_ctime).strftime('%Y년')
        category = get_category_from_ai(item.name, item.suffix)
        dest = BASE_DIR / year / category
        dest.mkdir(parents=True, exist_ok=True)
        shutil.move(str(item), str(dest / item.name))
        print(f"[{item.name}] → {year}/{category} ✅")

if __name__ == "__main__":
    if not TARGET_FOLDER.exists():
        TARGET_FOLDER.mkdir(parents=True, exist_ok=True)
    else:
        organize_and_move_files()

만들게 된 계기

FPS 장르는 총이 주력 무기인 게임이 대부분입니다. 하지만 냉병기 시대에도 활은 가장 보편적인 원거리 무기였습니다. "총이 없어도 FPS가 될 수 있지 않을까?" 라는 아이디어에서 출발해, 활의 차징과 포물선이라는 물리적 특성을 FPS에 접목한 1vs1 대전 게임을 개발하게 되었습니다.

게임 설명 & 핵심 기능

  • 🏹 차징 발사 시스템 — 마우스를 누르는 시간에 따라 화살 속도 변화 (Lerp 적용)
  • 🌀 포물선 물리 — Rigidbody + 중력으로 자연스러운 화살 궤적 구현
  • 🎯 부위별 데미지 배율 — 머리 175% / 몸통 100% / 팔다리 60%
  • 🌐 온라인 매칭 — Unity Lobby + Relay로 NAT 없이 P2P 매칭
  • 💊 힐 아이템 — 15초 쿨타임으로 리스폰되는 회복 아이템

핵심 알고리즘

  • ⚡ 차징 보간 — Mathf.Lerp(minSpeed, maxSpeed, chargeRatio)로 최소~최대 속도를 부드럽게 연결
  • 🔄 네트워크 동기화 — 20Hz로 위치/회전 전송 후 Vector3.Lerp로 상대방 캐릭터를 부드럽게 보간
  • 📡 매칭 파이프라인 — Lobby에서 빈 방 탐색 → 없으면 호스트 생성 → Relay JoinCode로 연결
  • 🎯 타격 판정 — ServerRpc로 타격 이벤트 전달 → 피격 클라이언트에서만 데미지 처리

핵심 코드

private void HandleBow()
{
    if (Input.GetMouseButtonDown(0))
    {
        isCharging = true;
        chargeTime = 0f;
    }

    if (Input.GetMouseButton(0) && isCharging)
    {
        chargeTime += Time.deltaTime;
        chargeTime = Mathf.Clamp(chargeTime, 0f, maxChargeTime);
    }

    if (Input.GetMouseButtonUp(0) && isCharging)
    {
        Shoot();
        isCharging = false;
        chargeTime = 0f;
    }
}

private void Shoot()
{
    float chargeRatio = chargeTime / maxChargeTime;
    // 차징 비율에 따라 속도 보간 (최소 ~ 최대)
    float arrowSpeed = Mathf.Lerp(minArrowSpeed, maxArrowSpeed, chargeRatio);

    Vector3 shootDirection = playerCamera.transform.forward;
    NetworkArrow networkArrow = GetComponent<NetworkArrow>();
    if (networkArrow != null)
        networkArrow.ShootArrowServerRpc(nockingPoint.position, shootDirection, arrowSpeed);
}

// 차징 비율 반환 (애니메이션, UI에서 활용)
public float GetChargeRatio() => isCharging ? chargeTime / maxChargeTime : 0f;
private void Update()
{
    if (!isLaunched) return;
    // 매 프레임 속도 방향으로 화살 회전 → 포물선 시각화
    if (rb.velocity.sqrMagnitude > 0.1f)
        transform.rotation = Quaternion.LookRotation(rb.velocity);
}

public void Launch(Vector3 direction, float speed)
{
    isLaunched = true;
    rb.isKinematic = false;
    // 초기 velocity 설정 → Rigidbody가 중력을 자동으로 더해 포물선 완성
    rb.velocity = direction * speed;
}

private void OnTriggerEnter(Collider other)
{
    if (!isLaunched) return;
    Hitbox hitbox = other.GetComponent<Hitbox>();
    if (hitbox != null)
    {
        NetworkObject netObj = other.GetComponentInParent<NetworkObject>();
        if (shooterNetworkArrow != null)
            shooterNetworkArrow.HitPlayerServerRpc(
                netObj.OwnerClientId, hitbox.GetHitboxTag(), damage);
    }
    StickToSurface();
}

private void StickToSurface()
{
    isLaunched = false;
    rb.isKinematic = true;  // 물리 비활성화 → 꽂힌 위치에 고정
    rb.velocity = Vector3.zero;
    Invoke("DestroySelf", 10f);
}
// 히트박스 태그에 따라 배율 적용
public void TakeDamage(float baseDamage, string hitboxTag)
{
    if (isDead || !isOwner) return;

    float multiplier = GetMultiplier(hitboxTag);
    float finalDamage = baseDamage * multiplier;

    currentHP -= finalDamage;
    currentHP = Mathf.Clamp(currentHP, 0f, maxHP);
    UpdateHPBar();

    if (currentHP <= 0f) Die();
}

private float GetMultiplier(string hitboxTag)
{
    switch (hitboxTag)
    {
        case "Head": return headMultiplier;   // 175%
        case "Body": return bodyMultiplier;   // 100%
        case "Limb": return limbMultiplier;   // 60%
        default:     return bodyMultiplier;
    }
}

private void UpdateHPBar()
{
    if (hpBarImage == null) return;
    hpBarImage.fillAmount = currentHP / maxHP;
    // HP 비율에 따라 빨강 ↔ 초록 그라데이션
    hpBarImage.color = Color.Lerp(Color.red, Color.green, currentHP / maxHP);
}
private void Update()
{
    if (IsOwner) HandleSend();     // 내 캐릭터 → 전송
    else         HandleReceive();  // 상대 캐릭터 → 보간
}

// 20Hz로 위치/회전 전송
private void HandleSend()
{
    sendTimer += Time.deltaTime;
    if (sendTimer < sendRate) return;  // sendRate = 0.05f (20Hz)
    sendTimer = 0f;
    SubmitPositionServerRpc(
        playerBody.position,
        playerBody.eulerAngles.y,
        playerCamera.localEulerAngles.x);
}

// 받은 목표값으로 부드럽게 이동 (Lerp 보간)
private void HandleReceive()
{
    playerBody.position = Vector3.Lerp(
        playerBody.position, targetPosition, lerpSpeed * Time.deltaTime);
    playerBody.rotation = Quaternion.Lerp(
        playerBody.rotation,
        Quaternion.Euler(0f, targetBodyRotY, 0f),
        lerpSpeed * Time.deltaTime);
}

[ServerRpc(RequireOwnership = true)]
private void SubmitPositionServerRpc(Vector3 pos, float bodyY, float camX)
    => BroadcastPositionClientRpc(pos, bodyY, camX);  // 서버 → 전체 전파

[ClientRpc]
private void BroadcastPositionClientRpc(Vector3 pos, float bodyY, float camX)
{
    if (IsOwner) return;  // 본인 제외
    targetPosition = pos;
    targetBodyRotY  = bodyY;
    targetCamRotX   = camX;
}
public async void StartMatching()
{
    if (IsMatching) return;
    IsMatching = true;

    // 1. 빈 Lobby 탐색 시도
    bool joined = await TryJoinExistingLobby();

    // 2. 없으면 내가 호스트로 생성
    if (!joined) await CreateLobby();
}

private async Task<bool> TryJoinExistingLobby()
{
    try
    {
        currentLobby = await LobbyService.Instance.QuickJoinLobbyAsync();
        // Lobby에 저장된 Relay JoinCode로 접속
        string relayCode = currentLobby.Data[RELAY_CODE_KEY].Value;
        await RelayManager.Instance.JoinRelay(relayCode);
        pollCoroutine = StartCoroutine(PollLobbyStatus());
        return true;
    }
    catch { return false; }
}

// 1.5초마다 인원 체크 → 2명 차면 게임 시작
private IEnumerator PollLobbyStatus()
{
    while (true)
    {
        yield return new WaitForSeconds(1.5f);
        var task = LobbyService.Instance.GetLobbyAsync(currentLobby.Id);
        yield return new WaitUntil(() => task.IsCompleted);
        currentLobby = task.Result;

        if (currentLobby.Players.Count >= MAX_PLAYERS)
        {
            StartGame();
            yield break;
        }
    }
}
